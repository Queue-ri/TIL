```java
class Solution {
    public int maxDistance(String s, int k) {
        char[] sArr = s.toCharArray();
        int N, S, E, W;
        N = S = E = W = 0;

        int ans = 0;
        for (int i = 0; i < sArr.length; ++i) {
            char ch = sArr[i];

            if (ch == 'N') ++N;
            else if (ch == 'S') ++S;
            else if (ch == 'E') ++E;
            else ++W;

            int d1 = Math.abs(N - S);
            int d2 = Math.abs(E - W);
            int md = d1 + d2;
            int cand = md + Math.min(2*k, i+1-md);
            ans = ans < cand ? cand : ans;
        }

        return ans;
    }
}
```

class Solution {
    public int[] minBitwiseArray(List<Integer> nums) {
        int[] ans = new int[nums.size()];
        for (int i = 0; i < nums.size(); ++i) {
            for (int x = 0; x < 1001; ++x) {
                if((x | (x+1)) == nums.get(i)) {
                    ans[i] = x;
                    break;
                }
                ans[i] = -1;
            }
        }

        return ans;
    }
}

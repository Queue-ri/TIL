def solution():
    n = int(input())
    crane = list(map(int, input().split()))
    m = int(input())

    crane.sort(reverse=True)
    box.sort(reverse=True)
    
    # corner case
    if crane[0] < box[0]:
        print(-1)
        return
    
    q = deque(box) # remained boxes
    ans = 0
    
    while q:
        #print(q)

        last_cdx = 0
        for _ in range(sz):
            if last_cdx > n-1:
                break
            
            b = q.popleft()
            if b <= crane[last_cdx]:
                last_cdx += 1
            else:
                q.append(b)
        
        ans += 1
    
    print(ans)
    
    
if __name__ == '__main__':
